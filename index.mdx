  <Steps>
    <Step title="Set up your database">
      Configure PostgreSQL connection string in your environment or `appsettings.json`:

      ```bash
      DATABASE_URL=postgresql://user:password@localhost:5432/task_manager
      ```

      Apply database migrations:

      ```bash
      dotnet ef database update
      ```
    </Step>

    <Step title="Start the backend">
      Navigate to the backend directory and run:

      ```bash
      cd backend
      dotnet run
      ```

      The API will be available at `https://localhost:7067`
    </Step>

    <Step title="Launch the frontend">
      Install dependencies and start the development server:

      ```bash
      npm install
      npm run dev
      ```

      Access the application at `http://localhost:8080`
    </Step>

    <Step title="Create your first task">
      Log in with your credentials and create a task:

      ```bash
      curl -X POST https://localhost:7067/api/tasks \
        -H "Authorization: Bearer YOUR_TOKEN" \
        -H "Content-Type: application/json" \
        -d '{
          "title": "My First Task",
          "description": "Getting started with task management",
          "priorityId": 1,
          "stateId": 1,
          "projectId": 1
        }'
      ```
    </Step>
  </Steps>
