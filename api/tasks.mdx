## Get All Tasks

<Card title="GET /api/Tasks" icon="list">
  Retrieve all tasks in the system
</Card>

### Headers

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

### Response

<ResponseField name="Success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="Message" type="string">
  Response message
</ResponseField>

<ResponseField name="Data" type="array">
  Array of task objects
  <Expandable title="Task object">
    <ResponseField name="Id" type="integer">
      Task ID
    </ResponseField>
    <ResponseField name="Title" type="string">
      Task title
    </ResponseField>
    <ResponseField name="Description" type="string">
      Task description
    </ResponseField>
    <ResponseField name="StateId" type="integer">
      State ID
    </ResponseField>
    <ResponseField name="StateName" type="string">
      State name
    </ResponseField>
    <ResponseField name="PriorityId" type="integer">
      Priority ID
    </ResponseField>
    <ResponseField name="PriorityName" type="string">
      Priority name
    </ResponseField>
    <ResponseField name="ProjectId" type="integer">
      Project ID
    </ResponseField>
    <ResponseField name="ProjectName" type="string">
      Project name
    </ResponseField>
    <ResponseField name="AsignedId" type="integer">
      Assigned user ID
    </ResponseField>
    <ResponseField name="AsignedName" type="string">
      Assigned user name
    </ResponseField>
    <ResponseField name="ExpirationDate" type="string">
      Task expiration date (ISO 8601 format)
    </ResponseField>
    <ResponseField name="EstimatedHours" type="integer">
      Estimated hours to complete
    </ResponseField>
    <ResponseField name="Usercreate" type="integer">
      User ID who created the task
    </ResponseField>
    <ResponseField name="Usermod" type="integer">
      User ID who last modified the task
    </ResponseField>
    <ResponseField name="CreationDate" type="string">
      Creation date (ISO 8601 format)
    </ResponseField>
    <ResponseField name="ModificationDate" type="string">
      Last modification date (ISO 8601 format)
    </ResponseField>
    <ResponseField name="Active" type="boolean">
      Whether the task is active
    </ResponseField>
  </Expandable>
</ResponseField>

### Example Request

```bash
curl -X GET https://api.example.com/api/Tasks \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

## Create Task

<Card title="POST /api/Tasks" icon="plus">
  Create a new task
</Card>

### Request Body

<ParamField body="Title" type="string" required>
  Task title
</ParamField>

<ParamField body="Description" type="string">
  Task description (optional)
</ParamField>

<ParamField body="StateId" type="integer" required>
  State ID for the task
</ParamField>

<ParamField body="PriorityId" type="integer" required>
  Priority ID for the task
</ParamField>

<ParamField body="ProjectId" type="integer" required>
  Project ID for the task
</ParamField>

<ParamField body="AsignedId" type="integer" required>
  User ID to assign the task to
</ParamField>

<ParamField body="ExpirationDate" type="string" required>
  Task expiration date (ISO 8601 format)
</ParamField>

<ParamField body="EstimatedHours" type="integer" default={0}>
  Estimated hours to complete the task
</ParamField>

### Example Request

```bash
curl -X POST https://api.example.com/api/Tasks \
  -H "Authorization: Bearer YOUR_JWT_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "Title": "Migrate database schema",
    "Description": "Update legacy database schema to new format",
    "StateId": 1,
    "PriorityId": 3,
    "ProjectId": 1,
    "AsignedId": 5,
    "ExpirationDate": "2026-04-15T00:00:00Z",
    "EstimatedHours": 40
  }'
```

### Example Response

<CodeGroup>
```json 201 Created
{
  "Success": true,
  "Message": "Task created successfully",
  "Data": {
    "Id": 1,
    "Title": "Migrate database schema",
    "Description": "Update legacy database schema to new format",
    "StateId": 1,
    "StateName": "To Do",
    "PriorityId": 3,
    "PriorityName": "High",
    "ProjectId": 1,
    "ProjectName": "Legacy Migration",
    "AsignedId": 5,
    "AsignedName": "John Doe",
    "ExpirationDate": "2026-04-15T00:00:00Z",
    "EstimatedHours": 40,
    "Usercreate": 1,
    "CreationDate": "2026-03-04T10:00:00Z",
    "Active": true
  }
}
```

```json 400 Bad Request
{
  "Success": false,
  "Message": "Invalid task data",
  "Data": null
}
```

```json 401 Unauthorized
"Usuario no válido"
```
</CodeGroup>
